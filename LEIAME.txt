# Projeto 5 - Grafo de Conhecimento com GraphRAG Para Aplicação de Análise de Contratos com IA

# Para executar a app:

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsaragp5 python=3.12

# Ative o ambiente:

conda activate dsaragp5 (ou: source activate dsaragp5)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute: 

streamlit run dsaprojeto5.py

# Perguntas de exemplo:

# Qual o Objeto do Contrato?
# Quais são as Obrigações do Contratante?
# Em caso de rescisão por descumprimento de qualquer obrigação, a parte inadimplente ficará sujeita ao pagamento de multa compensatória de qual valor?

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsaragp5 --all
